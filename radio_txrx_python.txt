import serial
import re
import serial.tools.list_ports
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from collections import deque

# 列出可用的串口设备。
ports_list = list(serial.tools.list_ports.comports())
if len(ports_list) <= 0:
    print("无串口设备。")
else:
    print("可用的串口设备如下：")
    for comport in ports_list:
        print(list(comport)[0], list(comport)[1])

# 打开串口。
ser = serial.Serial("COM4", 115200, timeout=1)  # 打开COM17，将波特率配置为115200，其余参数使用默认值
if ser.isOpen():  # 判断串口是否成功打开
    print("打开串口成功。")
    print(ser.name)  # 输出串口号
else:
    print("打开串口失败。")

# 读取串口输入信息并输出。
while True:
    com_input = ser.readlines(100)  # 读取串口输入信息
    if com_input:  # 如果读取结果非空，则输出
        print(com_input)
        numbers = re.findall(r'\d+\.\d+', str(com_input))
        number = [float(x) for x in numbers]
        result = [x / 10000 for x in number]
        print(result)
        y = result
        x = np.linspace(0, 2 * np.pi, len(y))
        # 绘制函数的图像
        plt.plot(x, y)
        plt.xlabel('x')
        plt.ylabel('y')
        plt.grid(True)
        # 显示图像
        plt.show()
